const form = document.getElementById("registerForm");
const registerBtn = document.getElementById("registerBtn");
 
const fullName = document.getElementById("fullName");
const email = document.getElementById("email");
const mobile = document.getElementById("mobile");
const password = document.getElementById("password");
const confirmPassword = document.getElementById("confirmPassword");
const dob = document.getElementById("dob");
const city = document.getElementById("city");
const terms = document.getElementById("terms");
 
const loader = document.getElementById("loader");
const successMessage = document.getElementById("successMessage");
 
const strengthBar = document.getElementById("strengthBar");
const strengthText = document.getElementById("strengthText");
 
const togglePassword = document.getElementById("togglePassword");
const themeToggle = document.getElementById("themeToggle");
 
// Regex Patterns
const namePattern = /^[A-Za-z ]+$/;
const emailPattern = /^[^ ]+@[^ ]+\.[a-z]{2,3}$/;
const mobilePattern = /^[0-9]{10}$/;
const passwordPattern = /^(?=.*[A-Z])(?=.*\d)(?=.*[@$!%*?&])[A-Za-z\d@$!%*?&]{8,}$/;
 
// Error Functions
function setError(id, message) {
  document.getElementById(id).innerText = message;
}
 
function setValid(input) {
  input.classList.add("valid");
  input.classList.remove("invalid");
}
 
function setInvalid(input) {
  input.classList.add("invalid");
  input.classList.remove("valid");
}
 
// Name Validation
function validateName() {
  if (!namePattern.test(fullName.value.trim())) {
    setError("nameError", "Enter a valid name");
    setInvalid(fullName);
    return false;
  }
 
  setError("nameError", "");
  setValid(fullName);
  return true;
}
 
// Email Validation
function validateEmail() {
  if (!emailPattern.test(email.value.trim())) {
    setError("emailError", "Enter a valid email address");
    setInvalid(email);
    return false;
  }
 
  setError("emailError", "");
  setValid(email);
  return true;
}
 
// Mobile Validation
function validateMobile() {
  if (!mobilePattern.test(mobile.value.trim())) {
    setError("mobileError", "Enter a 10-digit mobile number");
    setInvalid(mobile);
    return false;
  }
 
  setError("mobileError", "");
  setValid(mobile);
  return true;
}
 
// Password Validation
function validatePassword() {
  const value = password.value;
 
  if (value.length < 8) {
    strengthBar.style.width = "25%";
    strengthBar.style.background = "red";
    strengthText.innerText = "Weak Password";
 
    setError("passwordError", "Weak password");
    setInvalid(password);
    return false;
  }
 
  if (passwordPattern.test(value)) {
    strengthBar.style.width = "100%";
    strengthBar.style.background = "limegreen";
    strengthText.innerText = "Strong Password";
 
    setError("passwordError", "");
    setValid(password);
    return true;
  }
 
  strengthBar.style.width = "60%";
  strengthBar.style.background = "orange";
  strengthText.innerText = "Medium Password";
 
  setError(
    "passwordError",
    "Password needs uppercase, number & special character"
  );
 
  setInvalid(password);
  return false;
}
 
// Confirm Password
function validateConfirmPassword() {
  if (confirmPassword.value !== password.value || confirmPassword.value === "") {
    setError("confirmError", "Passwords do not match");
    setInvalid(confirmPassword);
    return false;
  }
 
  setError("confirmError", "");
  setValid(confirmPassword);
  return true;
}
 
// Gender Validation
function validateGender() {
  const gender = document.querySelector("input[name='gender']:checked");
 
  if (!gender) {
    setError("genderError", "Select your gender");
    return false;
  }
 
  setError("genderError", "");
  return true;
}
 
// DOB Validation
function validateDOB() {
  const birthDate = new Date(dob.value);
  const today = new Date();
 
  let age = today.getFullYear() - birthDate.getFullYear();
 
  if (!dob.value || age < 18) {
    setError("dobError", "You must be 18+");
    setInvalid(dob);
    return false;
  }
 
  setError("dobError", "");
  setValid(dob);
  return true;
}
 
// City Validation
function validateCity() {
  if (city.value === "") {
    setError("cityError", "Select your city");
    setInvalid(city);
    return false;
  }
 
  setError("cityError", "");
  setValid(city);
  return true;
}
 
// Terms Validation
function validateTerms() {
  if (!terms.checked) {
    setError("termsError", "Accept terms to continue");
    return false;
  }
 
  setError("termsError", "");
  return true;
}
 
// Enable Button Dynamically
function checkFormValidity() {
  const valid =
    validateName() &&
    validateEmail() &&
    validateMobile() &&
    validatePassword() &&
    validateConfirmPassword() &&
    validateGender() &&
    validateDOB() &&
    validateCity() &&
    validateTerms();
 
  registerBtn.disabled = !valid;
}
 
// Event Listeners
fullName.addEventListener("blur", validateName);
email.addEventListener("blur", validateEmail);
mobile.addEventListener("blur", validateMobile);
password.addEventListener("input", validatePassword);
confirmPassword.addEventListener("input", validateConfirmPassword);
dob.addEventListener("change", validateDOB);
city.addEventListener("change", validateCity);
terms.addEventListener("change", validateTerms);
 
// Auto Check Validity
form.addEventListener("input", checkFormValidity);
form.addEventListener("change", checkFormValidity);
 
// Show/Hide Password
togglePassword.addEventListener("click", () => {
  if (password.type === "password") {
    password.type = "text";
    togglePassword.innerHTML = '<i class="fa-solid fa-eye-slash"></i>';
  } else {
    password.type = "password";
    togglePassword.innerHTML = '<i class="fa-solid fa-eye"></i>';
  }
});
 
// Dark Mode Toggle
themeToggle.addEventListener("click", () => {
  document.body.classList.toggle("dark-mode");
});
 
// Auto Focus Next Field
fullName.addEventListener("keydown", (event) => {
  if (event.key === "Enter" && validateName()) {
    email.focus();
  }
});

 
// Form Submit
form.addEventListener("submit", (event) => {
  event.preventDefault();
 
  const isValid =
    validateName() &&
    validateEmail() &&
    validateMobile() &&
    validatePassword() &&
    validateConfirmPassword() &&
    validateGender() &&
    validateDOB() &&
    validateCity() &&
    validateTerms();
    
 
  if (!isValid) {
    return;
  }
 
  loader.style.display = "block";
  registerBtn.disabled = true;
 
  const userData = {
    fullName: fullName.value,
    email: email.value,
    mobile: mobile.value,
    gender: document.querySelector("input[name='gender']:checked").value,
    dob: dob.value,
    city: city.value
  };
 
  localStorage.setItem("smartTasksUser", JSON.stringify(userData));
 
  setTimeout(() => {
    loader.style.display = "none";
    successMessage.style.display = "block";
 
    form.reset();
 
    strengthBar.style.width = "0";
    strengthText.innerText = "Password Strength";
 
    document
      .querySelectorAll("input, select")
      .forEach((field) => {
        field.classList.remove("valid");
      });
 
    registerBtn.disabled = true;
 
  }, 2000);
});
